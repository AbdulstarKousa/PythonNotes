################################# General:
python --version
conda -V


################################# Stable environment with no dependency conflicts:
conda update --force conda
conda update anaconda
conda update conda


################################# Clean remove unused packages from the cache: 
conda clean --all
conda clean --force-pkgs-dirs # WARNING: This will break environments with packages installed using symlinks back to the package cache.


################################# Conda working with envs:
# List all environments:
conda env list
conda info --envs

# List all environment packages:
conda list

# Create (a data science) environment
conda create -n ds python=3.7.10 jupyterlab jupyter astroid babel pandas scipy seaborn scikit-learn

# Delete an environment
conda remove --name dsc --all

# Activate an environment:
conda activate myenv

# Deactivate an environment:
conda deactivate
