################################# General:
python --version    /* python version */
conda -V            /* conda version  */
-q                  /* after install to quit the print */ 
-- user             /* at the end of install command to suits your os */
conda install m2-base /*linux commands support*/

################################# Stable environment with no dependency conflicts:
conda update --force conda
conda update anaconda
conda update conda

################################# Clean remove unused packages from the cache: 
conda clean -a
conda clean --all
conda clean --force-pkgs-dirs # WARNING: This will break environments with packages installed using symlinks back to the package cache.


################################# Conda working with envs:
# List all environments:
conda env list
conda info --envs

# List all environment packages:
conda list

# Create (a data science) environment
conda create -n <mlops2022> python=3.7.10 jupyterlab jupyter astroid babel pandas scipy seaborn scikit-learn
conda activate <mlops2022>
conda install pytorch torchvision torchaudio cudatoolkit=11.3 -c pytorch


# Create with Clone and then update /* don't update the base */ 
conda create -n myenv --clone root
conda update -n myenv --all

# Delete an environment
conda remove --name dsc --all

# Activate an environment:
conda activate myenv

# Deactivate an environment:
conda deactivate

################################# Jupyter:
# Make sure that ipykernel is installed
pip install --user ipykernel

# Add the new virtual environment to Jupyter
python -m ipykernel install --user --name=dataScience

# List existing Jupyter virtual environments
jupyter kernelspec list

# List existing conda environments
conda env list

# Remove conda environment
conda env remove -n dsc

# Remove the environment from Jupyter
jupyter kernelspec uninstall dsc
